%% =====================================================
%% STRATEGIC PROCUREMENT PLANNING (RA 12009 EMPHASIS)
%% =====================================================

A[Identify Procurement Need] --> B{Included in APP?}

B -- No --> B1[Amend APP]
B1 --> B2{Approved by HOPE?}
B2 -- No --> B3[Return to End-User for Revision]
B2 -- Yes --> C[Proceed to Market Analysis]

B -- Yes --> C[Proceed to Market Analysis]

C --> D[Conduct Market Research / Industry Scan]
D --> E{Fit-for-Purpose Strategy Identified?}

E -- No --> D
E -- Yes --> F[Define Procurement Strategy]

F --> G{Method Selection}

G -- Competitive Bidding --> H[Prepare Bidding Documents]
G -- Alternative Method --> ALT0[Justify Alternative Method]
G -- Framework Agreement --> FA0[Initiate Framework Process]
G -- Emergency Procurement --> EM0[Document Emergency Grounds]

%% =====================================================
%% PRE-PROCUREMENT STAGE
%% =====================================================

H --> I[Pre-Procurement Conference]
I --> J{Complete & Compliant Documents?}
J -- No --> J1[Revise Documents]
J1 --> I
J -- Yes --> K[Approve Final Bidding Documents]

%% =====================================================
%% DIGITAL POSTING & ADVERTISEMENT
%% =====================================================

K --> L[Post Invitation (PhilGEPS / Electronic Portal)]
L --> M[Issue Bidding Documents]
M --> N[Conduct Pre-Bid Conference]

N --> O{Need Bid Bulletin?}
O -- Yes --> O1[Issue Supplemental/Bid Bulletin]
O1 --> P[Continue Bid Preparation]
O -- No --> P

%% =====================================================
%% BID SUBMISSION & OPENING
%% =====================================================

P --> Q[Receive Electronic / Physical Bids]
Q --> R{Number of Bids?}

R -- 0 --> R1[Declare Failure of Bidding]
R1 --> R2{Re-bid?}
R2 -- Yes --> L
R2 -- No --> ALT0

R -- 1 or more --> S[Public Bid Opening]
S --> T[Preliminary Examination]

T --> U{At least one passed Eligibility & Technical?}

U -- No --> U1[Declare Failure of Bidding]
U1 --> U2{Second Failure?}
U2 -- Yes --> ALT0
U2 -- No --> L

U -- Yes --> V[Detailed Evaluation]

%% =====================================================
%% EVALUATION (LCB/HRB or VALUE-BASED)
%% =====================================================

V --> W{Procurement Type?}

W -- Goods/Infra --> W1[Determine Lowest Calculated Bid]
W -- Consulting --> W2[Determine Highest Rated Bid]
W -- Value-Based Method --> W3[Apply Best Value Evaluation]

W1 --> X[Post-Qualification]
W2 --> X
W3 --> X

%% =====================================================
%% POST-QUALIFICATION & RISK CHECK
%% =====================================================

X --> Y{Passed Post-Qualification?}

Y -- No --> Y1[Disqualify Bidder]
Y1 --> Y2{Next Ranked Bid Available?}
Y2 -- Yes --> X
Y2 -- No --> Y3[Declare Failure]
Y3 --> Y4{Re-bid or Alternative?}
Y4 -- Re-bid --> L
Y4 -- Alternative --> ALT0

Y -- Yes --> Z[BAC Resolution Recommending Award]

%% =====================================================
%% APPROVAL & AWARD
%% =====================================================

Z --> AA{HOPE Approves?}

AA -- No --> AA1[Return to BAC for Review]
AA1 --> V

AA -- Yes --> AB[Issue Notice of Award]

AB --> AC{Submit Performance Security?}

AC -- No --> AC1[Forfeit Bid Security]
AC1 --> AC2{Next Ranked Bid?}
AC2 -- Yes --> X
AC2 -- No --> R1

AC -- Yes --> AD[Contract Signing]
AD --> AE[Issue Notice to Proceed]

%% =====================================================
%% CONTRACT IMPLEMENTATION & MONITORING
%% =====================================================

AE --> AF[Project Implementation]
AF --> AG{Need Variation / Change Order?}

AG -- Yes --> AG1[Evaluate Variation]
AG1 --> AG2{Within Legal Limits?}
AG2 -- No --> AG3[Seek Higher Approval or New Procurement]
AG2 -- Yes --> AF

AG -- No --> AH[Inspection & Acceptance]

AH --> AI{Accepted?}
AI -- No --> AI1[Require Rectification]
AI1 --> AF

AI -- Yes --> AJ[Process Payment]

%% =====================================================
%% PERFORMANCE & RISK MANAGEMENT
%% =====================================================

AJ --> AK{Performance Issues?}

AK -- Yes --> AK1[Issue Notice of Deficiency]
AK1 --> AK2{Severe Breach?}
AK2 -- Yes --> AK3[Terminate Contract]
AK3 --> AK4[Blacklisting Proceedings]
AK4 --> ALT0
AK2 -- No --> AF

AK -- No --> AL[Project Completion Report]

%% =====================================================
%% PROTEST & LEGAL REMEDY
%% =====================================================

Z --> PR{Bidder Files Protest?}

PR -- Yes --> PR1[HOPE Resolves Protest]
PR1 --> PR2{Granted?}
PR2 -- Yes --> V
PR2 -- No --> AB

PR -- No --> AB

%% =====================================================
%% ALTERNATIVE METHODS PATH
%% =====================================================

ALT0 --> ALT1{Grounds Valid & Approved?}
ALT1 -- No --> L
ALT1 -- Yes --> ALT2[Conduct Negotiated / Shopping / Direct Contracting]
ALT2 --> AD

%% =====================================================
%% FRAMEWORK AGREEMENT PATH
%% =====================================================

FA0 --> FA1[Pre-Qualification of Suppliers]
FA1 --> FA2[Framework Agreement Execution]
FA2 --> FA3[Call-Off Orders Issued]
FA3 --> AF

%% =====================================================
%% EMERGENCY PROCUREMENT PATH
%% =====================================================

EM0 --> EM1[Document Urgency Justification]
EM1 --> EM2[Direct Negotiation]
EM2 --> AD
